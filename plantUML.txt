package "org.example (Interfaces & Infrastructure)" {
    interface Positionable {
        + getPosition(): Point2D.Double
    }

    interface Movable {
        + move()
        + turnLeft()
        + turnRight()
    }

    interface Drivable {
        + getCurrentSpeed(): double
        + getEngineOn(): boolean
        + startEngine()
        + stopEngine()
        + gas(amount: double)
        + brake(amount: double)
    }

    interface Storable {
        + isStored(): boolean
        + unStore()
        + store(storage: Storage)
    }

    interface SmallCar
    interface BigCar

    interface Turbo {
        + setTurboOn()
        + setTurboOff()
    }

    interface Ramp {
        + setRampUp()
        + setRampDown()
        + getRampUp(): boolean
    }

    interface Storage<T extends Storable> {
        + load(item: T)
        + unload(): T
        + isEmpty(): boolean
        + isFull(): boolean
    }

    abstract class StorageUnit<T extends Storable> {
        - capacity: int
        - storage: ArrayList<T>
    }

    interface CarLoader<T extends SmallCar> {
        + tryToLoad(car: SmallCar)
        + getCarType(): Class<T>
    }

    class WorkShop<T extends SmallCar> {
        - carType: Class<T>
        - carBed: StorageUnit<T>
        - position: Point2D.Double
    }
}
